font properties

  • font-size:px/em/%/small;
  • font-family:"Arial";
  • font-color: slategray;
  • font-weight: bold;
  • font-style: bold;
  • font-variant: small-caps;

font shorthand

  • font: 16px "Arial";
  • font: 16px/1.5 "Arial";
  • font: 16px/1.5 "Arial", sans-serif;
  • font: 16px/1.5 "Helvetica Neue", Arial, sans-serif

% / em / rem

  • 100% === 16px
  • 1em === 16px
  • 1rem === 16px
